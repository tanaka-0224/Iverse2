import { useState, useEffect } from 'react';
import { User, Session } from '@supabase/supabase-js';
import { supabase } from '../lib/supabase';

const AUTH_SYNC_EVENT = 'app-auth-sync';
const hasWindow = () => typeof window !== 'undefined';
const broadcastAuthSync = (user: User | null) => {
  if (!hasWindow()) return;
  window.dispatchEvent(
    new CustomEvent<User | null>(AUTH_SYNC_EVENT, { detail: user }),
  );
};

export function useAuth() {
  const [user, setUser] = useState<User | null>(null);
  const [session, setSession] = useState<Session | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    console.log('[Auth] 1. useEffect: èªè¨¼æƒ…å ±ã®åˆæœŸãƒ­ãƒ¼ãƒ‰ã‚’é–‹å§‹'); // ğŸ’¡ é–‹å§‹ãƒ­ã‚°
    
    let mounted = true;
    const handleAuthSync = (event: Event) => {
      if (!mounted) return;
      const nextUser = (event as CustomEvent<User | null>).detail ?? null;
      setSession(null);
      setUser(nextUser);
      setLoading(false);
    };
    if (hasWindow()) {
      window.addEventListener(
        AUTH_SYNC_EVENT,
        handleAuthSync as EventListener,
      );
    }
    
    // 1. ã‚»ãƒƒã‚·ãƒ§ãƒ³ã®åˆå›å–å¾—
    supabase.auth.getSession().then(({ data: { session } }) => {
      console.log(`[Auth] 2. getSessionå®Œäº†: SessionãŒå­˜åœ¨ã™ã‚‹ã‹? ${!!session}`); // ğŸ’¡ å®Œäº†ãƒ­ã‚°
      if (mounted) {
        setSession(session);
        setUser(session?.user ?? null);
        setLoading(false);
      }
    }).catch(err => {
        console.error('[Auth] getSessionã‚¨ãƒ©ãƒ¼:', err); // ğŸ’¡ ã‚¨ãƒ©ãƒ¼ãƒ­ã‚°
        if (mounted) {
          setLoading(false);
        }
    });

    // 2. èªè¨¼çŠ¶æ…‹ã®ãƒªã‚¢ãƒ«ã‚¿ã‚¤ãƒ è³¼èª­
    const {
      data: { subscription },
    } = supabase.auth.onAuthStateChange(async (event, session) => {
      console.log(`[Auth] 3. AuthStateChangeã‚¤ãƒ™ãƒ³ãƒˆç™ºç”Ÿ: ${event}`); // ğŸ’¡ ç™ºç”Ÿãƒ­ã‚°
      
      if (mounted) {
        if (event !== 'INITIAL_SESSION') {
          setSession(session);
          setUser(session?.user ?? null);
          setLoading(false);
        }
        
        if (event === 'SIGNED_IN' && session?.user) {
          console.log('[Auth] 4. SIGNED_IN: createOrUpdateUserã‚’å®Ÿè¡Œ'); // ğŸ’¡ å‡¦ç†é–‹å§‹
          await createOrUpdateUser(session.user);
          console.log('[Auth] 5. createOrUpdateUserå®Œäº†'); // ğŸ’¡ å‡¦ç†å®Œäº†
        }
      }
    });

    return () => {
      mounted = false;
      subscription.unsubscribe();
      if (hasWindow()) {
        window.removeEventListener(
          AUTH_SYNC_EVENT,
          handleAuthSync as EventListener,
        );
      }
    };
  }, []);

  const signUp = async (email: string, password: string, name: string) => {
    const { data, error } = await supabase.auth.signUp({ 
      email, 
      password,
      options: {
        data: {
          name: name
        },
      }
    });
    if (error) throw error;
    return data;
  };

  const signIn = async (email: string, password: string) => {
    const { data, error } = await supabase.auth.signInWithPassword({ email, password });
    if (error) throw error;
    return data;
  };

  const signOut = async () => {
    try {
      const { error } = await supabase.auth.signOut();
      if (error) throw error;
    } catch (error) {
      console.error('[Auth] signOut error:', error);
    } finally {
      setSession(null);
      setUser(null);
      setLoading(false);
      broadcastAuthSync(null);
    }
  };

  return {
    user,
    session,
    loading,
    signUp,
    signIn,
    signOut,
  };
}
